// Relationships - Main Database
Ref: tbDocuments.MemberOf > tbFiles.Id
Ref: tbDocumentsData.IDF > tbDocuments.Id
Ref: tbDocumentsNotes.IDF > tbDocuments.Id
Ref: tbFileInfoContent.FileIDF > tbFiles.Id
Ref: tbFiles.MemberOf > tbFolders.Id
Ref: tbFilesData.IDF > tbFiles.Id
Ref: tbFilesNotes.IDF > tbFiles.Id
Ref: tbFoldersAddrAllocation.FolderIDF > tbFolders.Id
Ref: tbFoldersData.IDF > tbFolders.Id
Ref: tbFoldersNotes.IDF > tbFolders.Id

// Permission System Relationships
Ref: tbPermitAllocation.GroupIDF > tbPermitGroups.Id
Ref: tbPermits.GroupIDF > tbPermitGroups.Id
Ref: tbPermitUsers.GroupIDF > tbPermitGroups.Id


// Relationships - Address Database
Ref: ant_address.tbAddressAllocation.AddrID1 > ant_address.tbAddress.Id
Ref: ant_address.tbAddressAllocation.AddrID2 > ant_address.tbAddress.Id
Ref: ant_address.tbAddressAllocation.AllocType > ant_address.tbAddressAllocationType.ID
Ref: ant_address.tbAddressCommunication.ComIDF > ant_address.tbAddressComType.ID
Ref: ant_address.tbAddressCommunication.AddrIDF > ant_address.tbAddress.Id
Ref: ant_address.tbAddressGroupAllocation.AdrIDF > ant_address.tbAddress.Id
Ref: ant_address.tbAddressGroupAllocation.GrpIDF > ant_address.tbAddressGroups.Id


// Cross-Database Relationship (Main DB to Address DB)
Ref: tbFoldersAddrAllocation.AddrIDF > ant_address.tbAddress.Id
Ref: tbFoldersAddrAllocation.Addr2ID > ant_address.tbAddress.Id
